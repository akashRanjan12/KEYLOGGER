cmake_minimum_required(VERSION 3.15)
project(EducationalKeylogger)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(EducationalKeylogger
    src/main.cpp
    src/keylogger.cpp
    src/security.cpp
)

target_include_directories(EducationalKeylogger PRIVATE include)
target_link_libraries(EducationalKeylogger User32)

# Compiler flags for security
if(MSVC)
    target_compile_options(EducationalKeylogger PRIVATE /W4 /sdl)
else()
    target_compile_options(EducationalKeylogger PRIVATE -Wall -Wextra -Wpedantic)
endif()
